Memasukkan Input dengan variable A, B, C
-----------

-----------
Memasukkan Input dengan array dengan nama variable M dengan panjang N
-----------
ll N;
cin >> N;
ll M[N];
for (ll i = 0; i < N; i++) {
  cin >> M[i];
}

-----------

  cout << i << endl;
  i = i * 2;
